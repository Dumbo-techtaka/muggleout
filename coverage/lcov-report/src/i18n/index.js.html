function getSystemLocale() {
  const locale = process.env.LANG || process.env.LANGUAGE || process.env.LC_ALL || '';
  
  if (locale.includes('ko') || locale.includes('KR')) return 'ko';
  if (locale.includes('ja') || locale.includes('JP')) return 'ja';
  return 'en';
}
